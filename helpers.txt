import functions_framework
import json
from google.cloud import storage

@functions_framework.http
def hello_http(request):
    """HTTP Cloud Function that retrieves a value from a JSON file in Cloud Storage.
    Args:
        request (flask.Request): The request object.
    Returns:
        A tuple containing the response body and status code.
    """
    # Get parameters from the request
    request_json = request.get_json(silent=True)
    request_args = request.args

    # Set default values
    bucket_name = 'quiz-info'  # Replace with your actual bucket name
    file_name = 'innovators_final.json'
    key_to_access = 'George Carruthers'

    # Check if 'name' parameter is provided in the request
    if request_json and 'name' in request_json:
        key_to_access = request_json['name']
    elif request_args and 'name' in request_args:
        key_to_access = request_args['name']

    try:
        # Initialize the Cloud Storage client
        storage_client = storage.Client()

        # Get the bucket and blob
        bucket = storage_client.get_bucket(bucket_name)
        blob = bucket.blob(file_name)

        # Download the contents of the blob as a string and parse JSON
        json_data = json.loads(blob.download_as_text())

        # Access the requested key
        if key_to_access in json_data:
            return json.dumps(json_data[key_to_access]), 200
        else:
            return json.dumps({'error': f'Key "{key_to_access}" not found in the JSON data.'}), 404

    except Exception as e:
        return json.dumps({'error': str(e)}), 500
